`String.trim/1` strips both leading and trailing whitespace. So the production output

A heredoc's literal value INCLUDES a trailing newline AND optionally a leading newline
depending on how the opening `"""` is positioned. The `String.trim_trailing/1` applied
in `exact/1` removes only the trailing newline. The production `String.trim/1` removes
both leading and trailing whitespace. So for the test to pass, the heredoc must NOT
have a leading newline — i.e., the heredoc's opening triple-quote and the first content
line are tight (no blank line between them).

      @moduledoc """
      Aggregates Rindle runtime capability surfaces for ops/doctor consumers.

      Phase 33 ships the aggregator function only. Phase 36 (MUX-16) refactors
      `mix rindle.doctor` to consume `report/0`.

      @type report :: %{
              storage: %{module() => [atom()]},
              processor: %{module() => [atom()]},
              streaming: %{
                providers: %{module() => [atom()]},
                signed_playback_configured?: boolean(),
                configured_profiles: [module()]
              }
            }

      @spec report() :: report()
      def report do
        %{
          storage: storage_report(),
          processor: processor_report(),
          streaming: %{
            providers: streaming_providers_report(),
            signed_playback_configured?: signed_playback_configured?(),
            configured_profiles: configured_streaming_profiles()
          }
        }
      end

      # --- storage ---

      defp storage_report do
        for profile <- profile_modules(), into: %{} do
          adapter = safely_call_zero(profile, :storage_adapter)
          {adapter || profile, capabilities_for(adapter, Rindle.Storage.Capabilities)}
        end
      end

      # --- processor ---

      defp processor_report do
        # Phase 33 has no Rindle.Processor.Capabilities module yet (AV uses a
        # different shape via Rindle.AV.Capability). Map known processor modules to
        # their advertised capabilities. If the function is missing, return [].
        for profile <- profile_modules(), into: %{} do
          processor = safely_call_zero(profile, :processor)
          caps = if processor, do: safely_call_zero(processor, :capabilities) || [], else: []
          {processor || profile, List.wrap(caps)}
        end
      end

      # --- streaming ---

      defp streaming_providers_report do
        for profile <- profile_modules(),
            streaming = streaming_config_for(profile),
            not is_nil(streaming),
            provider = streaming.provider,
            not is_nil(provider),
            into: %{} do
          {provider, capabilities_for(provider, Rindle.Streaming.Capabilities)}
        end
      end

      defp signed_playback_configured? do
        cfg = Application.get_env(:rindle, Rindle.Streaming.Provider.Mux, [])

        is_binary(cfg_get(cfg, :signing_key_id)) and
          is_binary(cfg_get(cfg, :signing_private_key))
      end

      defp configured_streaming_profiles do
        for profile <- profile_modules(),
            not is_nil(streaming_config_for(profile)) do
          profile
        end
      end

      # --- helpers ---

      defp profile_modules do
        # Reuse existing config helper if available.
        if function_exported?(Rindle.Config, :profile_modules, 0) do
          Rindle.Config.profile_modules()
        else
          []
        end
      end

      defp streaming_config_for(profile) do
        case safely_call_zero(profile, :delivery_policy) do
          %{streaming: streaming} -> streaming
          _ -> nil
        end
      end

      defp capabilities_for(nil, _vocab), do: []

      defp capabilities_for(module, vocab) do
        if function_exported?(module, :capabilities, 0) do
          vocab.safe(module)
        else
          []
        end
      end

      defp safely_call_zero(module, fun) do
        cond do
          is_nil(module) -> nil
          function_exported?(module, fun, 0) -> apply(module, fun, [])
          true -> nil
        end
      rescue
        _ -> nil
      end

      defp cfg_get(cfg, key) when is_list(cfg), do: Keyword.get(cfg, key)
      defp cfg_get(cfg, key) when is_map(cfg), do: Map.get(cfg, key)
      defp cfg_get(_, _), do: nil
    end
