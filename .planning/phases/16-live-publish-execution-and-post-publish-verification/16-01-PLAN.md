```bash
#!/usr/bin/env bash
# Usage: bash test/install_smoke/support/fake_hex_bin.sh <shim_dir>
# Writes <shim_dir>/mix and <shim_dir>/curl that honor:
#   RINDLE_PROBE_FAKE_HEX_INFO_EXIT     -> mix hex.info exits with this (default 0)
#   RINDLE_PROBE_FAKE_HEX_INFO_STDOUT   -> mix hex.info prints this to stdout (default empty)
#   RINDLE_PROBE_FAKE_CURL_HTTP_STATUS  -> curl emits this body when invoked with -w '%{http_code}' (default 200)
#   RINDLE_PROBE_FAKE_CURL_EXIT         -> curl exit code (default 0)
set -euo pipefail
shim_dir="${1:?shim dir required}"
mkdir -p "$shim_dir"

cat > "$shim_dir/mix" <<'MIXEOF'
#!/usr/bin/env bash
# Intercept "hex.info <pkg> <version>"; exec real mix for everything else if RINDLE_REAL_MIX is set.
if [ "${1:-}" = "hex.info" ]; then
  printf '%s' "${RINDLE_PROBE_FAKE_HEX_INFO_STDOUT:-}"
  exit "${RINDLE_PROBE_FAKE_HEX_INFO_EXIT:-0}"
fi
# For non-intercepted invocations, delegate to real mix if available; otherwise fail loud.
if [ -n "${RINDLE_REAL_MIX:-}" ] && [ -x "$RINDLE_REAL_MIX" ]; then
  exec "$RINDLE_REAL_MIX" "$@"
fi
echo "fake mix shim: unhandled invocation: $*" >&2
exit 99
MIXEOF
chmod +x "$shim_dir/mix"

cat > "$shim_dir/curl" <<'CURLEOF'
#!/usr/bin/env bash
status="${RINDLE_PROBE_FAKE_CURL_HTTP_STATUS:-200}"
exit_code="${RINDLE_PROBE_FAKE_CURL_EXIT:-0}"
# Emulate curl -w '%{http_code}' contract used in the probe (write status to stdout when asked).
case " $* " in
  *" -w "*"%{http_code}"*) printf '%s' "$status" ;;
  *) ;;
esac
exit "$exit_code"
CURLEOF
chmod +x "$shim_dir/curl"
```

Create `scripts/hex_release_exists.sh` (chmod +x). Exact shape:

```bash
#!/usr/bin/env bash
# Idempotency probe — answers "is rindle <VERSION> already on Hex.pm?"
# Primary probe: mix hex.info rindle <VERSION>   (exit 0 = published)
# Fallback probe: curl https://hex.pm/api/packages/rindle/releases/<VERSION>  (HTTP 200 = published)
# Defense-in-depth (D-09): both probes consulted; if mix exec fails, fall through to curl.
# Output contract (Pitfall 4): exactly one trailing stdout line `already_published=(true|false)`.
# Auth ban (Pitfall 3): MUST NOT call mix hex.user, mix hex.owner, or any other auth-required Hex command.
# All diagnostics go to stderr.
set -euo pipefail

ROOT_DIR="${RINDLE_PROJECT_ROOT:-$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)}"
cd "$ROOT_DIR"

if [ "${RINDLE_PROBE_DEBUG:-0}" = "1" ]; then
  echo "hex_release_exists: cd: $(pwd)" >&2
fi

VERSION="${VERSION:?VERSION env var required (e.g., 0.1.4)}"

write_output() {
  # $1 = "true" or "false"
  echo "already_published=$1"
  if [ -n "${GITHUB_OUTPUT:-}" ]; then
    echo "already_published=$1" >> "$GITHUB_OUTPUT"
  fi
}

# --- Primary probe: mix hex.info rindle <VERSION> ---
mix_exit=0
mix hex.info rindle "$VERSION" >/dev/null 2>&1 || mix_exit=$?

if [ "$mix_exit" -eq 0 ]; then
  echo "hex_release_exists: mix hex.info rindle $VERSION exited 0 (published)" >&2
  write_output "true"
  exit 0
fi

# Distinguish "definitely not published" from "transient mix error".
# mix hex.info exits 1 for both missing version and transient errors.
# We always fall through to curl; the disambiguation happens via curl HTTP status.

# --- Fallback probe: curl https://hex.pm/api/packages/rindle/releases/<VERSION> ---
curl_exit=0
http_status=$(curl -sS -o /dev/null -w '%{http_code}' \
  "https://hex.pm/api/packages/rindle/releases/$VERSION") || curl_exit=$?

if [ "$curl_exit" -ne 0 ]; then
  echo "::error::hex_release_exists: both probes inconclusive — mix hex.info exited $mix_exit AND curl https://hex.pm/api/packages/rindle/releases/$VERSION failed (curl exit $curl_exit). Refusing to emit already_published — failing loud to avoid false-negative re-trigger of run 25135467509." >&2
  exit 1
fi

case "$http_status" in
  200)
    echo "hex_release_exists: curl returned HTTP 200 — fallback says published" >&2
    write_output "true"
    exit 0
    ;;
  404)
    if [ "$mix_exit" -eq 1 ]; then
      echo "hex_release_exists: mix exit 1 + curl 404 — version not published" >&2
      write_output "false"
      exit 0
    fi
    # mix had a non-1 exit (e.g., 127 binary missing) — but curl says 404 → trust 404.
    echo "hex_release_exists: mix exit $mix_exit + curl 404 — version not published" >&2
    write_output "false"
    exit 0
    ;;
  *)
    echo "::error::hex_release_exists: both probes inconclusive — mix hex.info exited $mix_exit AND curl returned HTTP $http_status (expected 200 or 404). Refusing to emit already_published." >&2
    exit 1
    ;;
esac
```
